Prompt -> And_or Prompt | epsilon

And_or -> Pipeline And_or''
	And_or'' -> And_or' And_or'' | epsilon
		And_or' -> || New_line Pipeline | && New_line Pipeline

New_line -> \n New_line' | epsilon
	New_line' -> \n New_line' | epsilon

Pipeline -> Cmd Pipeline'
	Pipeline' -> pipe New_line Cmd Pipeline' | epsilon

Cmd -> Parentheses | Prefixe Cmd'' | word Cmd'
	Cmd'' -> word Cmd' | epsilon
		Cmd' -> Suffixe | epsilon

Parentheses -> ( And_or )

Prefixe -> I/O_Redirection Prefixe'
	Prefixe' -> I/O_Redirection Prefixe' | epsilon

Suffixe -> word Suffixe'' | I/O_Redirection Suffixe''
	Suffixe'' -> Suffixe' Suffixe'' | epsilon
		Suffixe' -> I/O_Redirection | word

I/O_Redirection -> < word | > word | >> word | << word
